// 1. Firebase और PeerJS सेटअप const peer = new Peer(); // अपनी रैंडम आईडी जनरेट करेगा const db = firebase.database(); // फायरबेस कनेक्शन // 2. रैंडम प्लेयर खोजना function findRandomOpponent() { showStatus("खिलाड़ी की तलाश जारी है..."); const playersRef = db.ref('online_players'); // पहले से इंतज़ार कर रहे खिलाड़ियों को देखें playersRef.once('value', (snapshot) => { let found = false; snapshot.forEach((child) => { const player = child.val(); if (player.status === 'waiting' && child.key !== peer.id) { // मैच मिल गया! connectToOpponent(child.key); found = true; return true; } }); if (!found) { // कोई नहीं मिला, तो खुद "waiting" लिस्ट में जुड़ें playersRef.child(peer.id).set({ status: 'waiting', name: 'Player_' + Math.floor(Math.random() * 100) }); // जब कोई दूसरा हमसे कनेक्ट करेगा तो यह ट्रिगर होगा peer.on('connection', (conn) => { setupGame(conn); playersRef.child(peer.id).remove(); // लिस्ट से हट जाएं }); } }); } // 3. दूसरे प्लेयर से हाथ मिलाना (Voice + Data) async function connectToOpponent(targetId) { const conn = peer.connect(targetId); setupGame(conn); // माइक एक्सेस और वॉइस कॉल try { const stream = await navigator.mediaDevices.getUserMedia({ audio: true }); const call = peer.call(targetId, stream); call.on('stream', (remoteStream) => { const audio = new Audio(); audio.srcObject = remoteStream; audio.play(); showStatus("वॉइस चैट चालू है! 🎙️"); }); } catch (err) { console.error("माइक नहीं मिला:", err); } } // 4. गेम के अंदर बातचीत और स्कोर सिंक function setupGame(connection) { document.getElementById('countdown-screen').classList.remove('hidden'); connection.on('data', (data) => { if(data.type === 'kill') { s2 = data.score; document.getElementById('score2').innerText = s2; // जब वो मच्छर मारे, तो आपकी स्क्रीन पर भी '💥' दिखे showSplatOnOpponentAction(data.x, data.y); } }); }
